//曼哈顿距离法:
#include <iostream>
#include <cmath>

using namespace std;

int main(){
    int n;
    while(cin >> n, n ){

        for (int i = 0; i < n; i ++){
            for (int j = 0; j < n; j ++){
                if (n % 2) 
                    cout << (n + 1) / 2 - max(abs(n / 2 - i), abs(n / 2 - j)) << ' ';
                else
                    cout << (n + 1) / 2.0 - max(abs((n - 1) / 2.0 - i), abs((n - 1) / 2.0 - j)) << ' ';
            }
            cout << endl;
        }

        cout << endl;
    }

    return 0;
}
