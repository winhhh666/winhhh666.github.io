typedef struct Node {

node* initList(int n) {
    node* head = (node*)malloc(sizeof(node));
    node* temph = head;
    node* temp;
    for(int i = 0; i<n; i++) {
        temp = (node*)malloc(sizeof(node));
        scanf("%d %d", &temp->num, &temp->grade);
        temp->next = NULL;
        temph->next = temp;
        temph = temp->next;
    }
    return head;
}

node* mergList(node* a, node* b, int all) {
    node* tempa = a;
    while(tempa->next != NULL) {
        tempa = tempa->next;
    }
    tempa->next = b->next;
    node* tempb = b;
    node* temp;
    node* nmin;
    int imin;
    for(int i=0; i < all; i++) {
        nmin = a->next;
        imin = a->next->num;
        temp = a;
        while(temp->next!= NULL) {
            temp = temp->next;
            if(temp->num < imin) {
                imin = temp->num;
                nmin = temp;
            }
        }
        tempb->next = temp;
        tempb = tempb->next;
    } 
    return b;
}

int main() {
    int m, n;
    node *a, *b, *ab;
    scanf("%d %d\n", &m, &n);
    a = initList(m);
    b = initList(n);
    ab = mergList(a, b, m+n);
    for(int i = 0; i<m+n; i++){
        ab = ab->next;
        printf("%d %d\n", ab->num, ab->grade);
    }
}

line* insertLine(line* head, int data, int add) {
    //三个参数分别为:进行此操作的双链表,插入数据,插入位置
    //新建数据域data的结点
    line* temp = (line*)malloc(sizeof(line));
    temp->data = data;
    temp->pre = NULL;
    temp->enxt = NULL;
    //插入到链表头,要特殊考虑
    if(add == 1) {
        temp->next = head;
        head->pre = temp;
        head = temp;
    }else {
        line* body = head;
        //找到要插入位置的前一个结点
        for(int i=1; i<add-1; i++) {
            body = body->next;
        }
        if(body->next == NULL) {
            body->next = temp;
            temp->pre = body;
        }else{
            body->next->pre = temp;
            temp->next = body->next;
            body->next = temp;
            temp->pre = body;
        }
    }
    return head;
}

        return NULL;
    Node *temp;

void pop(queue *q){
    node* n = q->front;

        return ;

        q->front = NULL;
        q->rear = NULL;

    }else{
        q->front = q->front->next;

}

循环队列

#define maxsize 10

typedef struct cir_queue{
    int data[maxsize];
    int rear;
    int front;
}cir_queue;
//入队操作
void push(cir_queue *q, int data){
    if((q->rear+1)%maxsize == q->front){
        printf("溢出,无法入队\n");
        return ;
    }else{
        q->rear = (q->rear+1)%maxsize;
        q->data[q->rear] = data;
    }
}
//出队操作
void pop(cir_queue *q){
    if(q->rear == q->front){
        printf("队列为空, 无法出队\n");
        return;
    }else{
        q->data[q->front] = 0;
        q->front = (q->front+1)%maxsize;
    }
}

{
    int value;
    int dir;
};

int StackFib(int n)
{
    node w;
    stack<node> s;
    int sum = 0;
    
    while(1)
    {
        while(1){
            w.value = n;
            w.dir = 1;
            s.push(w);
            n--;
            if(n<0)
                break;
        }
        while(!s.empty())
        {
            w=s.top();
            s.pop();
            if(w.value == 0)
                sum += 0;
            else if(w.value == 1)
                sum+=1;
            else if(w.value != 0 && w.value != 1 && w.dir == 1)
            {
                w.dir =  2;
                s.push(w);
                n = w.value-2;
                break;
            }
        }
        if(s.empty())
            break;
    }
    return sum;
}
